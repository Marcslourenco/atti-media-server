```python
class EventType(str, Enum):
    INTRO = "intro"
    USER_QUERY = "query"

@app.post("/api/avatar/speak")
async def speak(
    avatar_id: str,
    text: str,
    language: str = "pt-BR",
    event_type: EventType = EventType.USER_QUERY,
    session_id: Optional[str] = None,
):
    if event_type == EventType.INTRO:
        intro_text = f"Olá! Sou {avatar_id.capitalize()}. Como posso ajudar?"
        audio_data = await tts_generate(intro_text, avatar_id, language)
        return {"response": intro_text, "audio": audio_data, "source": "intro"}

    # Pipeline normal (RAG + LLM)
    response = await rag_engine.generate_response(
        query=text,
        avatar_id=avatar_id,
        session_id=session_id
    )
    return response
```

### 2.1 Modificar worker_ingest_buildtime.py

Substitua o parser existente por:

```python
import re
import logging

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")

def parse_qa_document(text: str, avatar_id: str) -> list[dict]:
    """Parser para documentos Q/A: resposta = conteúdo, pergunta = metadado."""
    chunks = []
    pattern = r'Q:\s*(.+?)\s*A:\s*(.+?)(?=Q:|$)'
    matches = re.findall(pattern, text, re.DOTALL)
    for question, answer in matches:
        answer_clean = answer.strip()
        if answer_clean:
            chunks.append({
                "content": answer_clean,
                "metadata": {
                    "avatar_id": avatar_id,
                    "question_trigger": question.strip(),
                    "doc_type": "qa"
                }
            })
    logging.info(f"[Parser Q/A] avatar={avatar_id}: {len(chunks)} chunks extraídos")
    return chunks

def parse_narrative_document(text: str, avatar_id: str) -> list[dict]:
    """Parser para documentos narrativos com overlap."""
    chunk_size = 400
    overlap = 60
    chunks = []
    for i in range(0, len(text), chunk_size - overlap):
        chunk = text[i:i + chunk_size].strip()
        if chunk:
            chunks.append({
                "content": chunk,
                "metadata": {
                    "avatar_id": avatar_id,
                    "doc_type": "narrative"
                }
            })
    logging.info(f"[Parser Narrativo] avatar={avatar_id}: {len(chunks)} chunks extraídos")
    return chunks

def ingest_document(text: str, avatar_id: str, doc_type: str = "auto") -> list[dict]:
    """Dispatch automático de parser."""
    if doc_type == "auto":
        doc_type = "qa" if re.search(r'Q:\s*.+\s*A:\s*', text, re.DOTALL) else "narrative"
    if doc_type == "qa":
        return parse_qa_document(text, avatar_id)
    return parse_narrative_document(text, avatar_id)
```

### 3.2 Implementar BrainManager em src/brain_manager.py

```python
import os
import logging
from pathlib import Path

class BrainManager:
    def __init__(self, base_path: str = "knowledge"):
        self.base_path = Path(base_path)
        self._cache = {}
        self._load_all()

    def _load_all(self):
        for avatar_dir in self.base_path.iterdir():
            if not avatar_dir.is_dir():
                continue
            avatar_id = avatar_dir.name
            prompts_dir = avatar_dir / "prompts"
            if not prompts_dir.exists():
                logging.warning(f"[BrainManager] Diretório prompts não encontrado: {prompts_dir}")
                continue
            self._cache[avatar_id] = {}
            for f in prompts_dir.glob("*.txt"):
                key = f.stem
                self._cache[avatar_id][key] = f.read_text(encoding="utf-8").strip()
                logging.info(f"[BrainManager] Carregado: {avatar_id}/{key} ({len(self._cache[avatar_id][key])} chars)")

    def get_system_prompt(self, avatar_id: str) -> str:
        return self._cache.get(avatar_id, {}).get("system_prompt", "")

    def get_rules(self, avatar_id: str) -> str:
        return self._cache.get(avatar_id, {}).get("regras_conversacao", "")

    def get_variations(self, avatar_id: str) -> str:
        return self._cache.get(avatar_id, {}).get("variacao_respostas", "")
```

### 4.1 Criar src/llm_orchestrator.py

```python
import httpx
import re
import os
import logging

OLLAMA_URL = os.getenv("OLLAMA_URL", "http://localhost:11434")
OLLAMA_MODEL = os.getenv("OLLAMA_MODEL", "phi3:mini")

# Fallback gratuito via OpenRouter (Qwen2.5 — sem custo)
OPENROUTER_API_KEY = os.getenv("OPENROUTER_API_KEY", "")  # chave gratuita do openrouter.ai
OPENROUTER_MODEL = os.getenv("OPENROUTER_MODEL", "qwen/qwen-2.5-7b-instruct:free")

async def _call_ollama(system_prompt: str, messages: list, user_query: str, context_docs: str) -> str:
    payload = {
        "model": OLLAMA_MODEL,
        "messages": [
            {"role": "system", "content": system_prompt},
            *messages[-6:],
            {"role": "user", "content": f"Documentos relevantes:\n{context_docs}\n\nPergunta: {user_query}"}
        ],
        "stream": False
    }
    async with httpx.AsyncClient(timeout=15.0) as client:
        resp = await client.post(f"{OLLAMA_URL}/api/chat", json=payload)
        resp.raise_for_status()
        return resp.json()["message"]["content"].strip()

async def _call_openrouter(system_prompt: str, messages: list, user_query: str, context_docs: str) -> str:
    if not OPENROUTER_API_KEY:
        raise ValueError("OPENROUTER_API_KEY não configurado")
    headers = {
        "Authorization": f"Bearer {OPENROUTER_API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": OPENROUTER_MODEL,
        "messages": [
            {"role": "system", "content": system_prompt},
            *messages[-6:],
            {"role": "user", "content": f"Documentos relevantes:\n{context_docs}\n\nPergunta: {user_query}"}
        ]
    }
    async with httpx.AsyncClient(timeout=20.0) as client:
        resp = await client.post("https://openrouter.ai/api/v1/chat/completions", headers=headers, json=payload)
        resp.raise_for_status()
        return resp.json()["choices"][0]["message"]["content"].strip()

def _rag_fallback(context_docs: str) -> str:
    """Fallback interno: limpa e retorna o contexto RAG diretamente."""
    clean = re.sub(r'^Q:.*?\n', '', context_docs, flags=re.MULTILINE)
    clean = re.sub(r'^A:\s*', '', clean, flags=re.MULTILINE)
    return clean.strip()[:300]

async def generate_llm_response(
    system_prompt: str,
    context_docs: str,
    history: list,
    user_query: str
) -> dict:
    """
    Hierarquia de LLMs (100% gratuitos):
    1. Ollama local (phi3:mini ou qwen2.5:1.5b)
    2. OpenRouter Qwen free tier (requer OPENROUTER_API_KEY gratuita)
    3. Fallback RAG interno (sem LLM externo)
    """
    # Tentativa 1: Ollama
    try:
        response = await _call_ollama(system_prompt, history, user_query, context_docs)
        logging.info("[LLM] Resposta via Ollama")
        return {"response": response, "source": "ollama"}
    except Exception as e:
        logging.warning(f"[LLM] Ollama falhou: {e}")

    # Tentativa 2: OpenRouter (Qwen free)
    try:
        response = await _call_openrouter(system_prompt, history, user_query, context_docs)
        logging.info("[LLM] Resposta via OpenRouter/Qwen (gratuito)")
        return {"response": response, "source": "openrouter_free"}
    except Exception as e:
        logging.warning(f"[LLM] OpenRouter falhou: {e}")

    # Tentativa 3: Fallback RAG interno
    response = _rag_fallback(context_docs)
    logging.info("[LLM] Usando fallback RAG interno")
    return {"response": response, "source": "rag_fallback"}
```

### 5.1 Implementar SessionMemory com fallback in-memory

```python
import os
import json
import logging
from typing import Optional

# Tenta usar Redis se configurado, senão usa dict em memória
try:
    import redis.asyncio as aioredis
    REDIS_URL = os.getenv("REDIS_URL", "")
    _use_redis = bool(REDIS_URL)
except ImportError:
    _use_redis = False

_memory_store = {}  # fallback in-memory

class SessionMemory:
    def __init__(self, session_id: str, max_turns: int = 6):
        self.session_id = session_id
        self.max_turns = max_turns
        self._redis = None

    async def _get_redis(self):
        if _use_redis and self._redis is None:
            self._redis = await aioredis.from_url(os.getenv("REDIS_URL"))
        return self._redis

    async def get_history(self) -> list:
        if _use_redis:
            try:
                r = await self._get_redis()
                data = await r.get(f"session:{self.session_id}")
                return json.loads(data) if data else []
            except Exception as e:
                logging.warning(f"[Memory] Redis falhou, usando in-memory: {e}")
        return _memory_store.get(self.session_id, [])

    async def add_turn(self, user_msg: str, assistant_msg: str):
        history = await self.get_history()
        history.append({"role": "user", "content": user_msg})
        history.append({"role": "assistant", "content": assistant_msg})
        history = history[-(self.max_turns * 2):]  # mantém últimos N turnos

        if _use_redis:
            try:
                r = await self._get_redis()
                await r.setex(f"session:{self.session_id}", 3600, json.dumps(history))
                return
            except Exception as e:
                logging.warning(f"[Memory] Redis falhou ao salvar: {e}")
        _memory_store[self.session_id] = history

    async def clear(self):
        if _use_redis:
            try:
                r = await self._get_redis()
                await r.delete(f"session:{self.session_id}")
            except:
                pass
        _memory_store.pop(self.session_id, None)
```

**Nota:** O fallback in-memory perde sessões ao reiniciar o container. É aceitável para MVP. Quando quiser persistência, adicione o Redis add-on do Render ($3/mês) e configure `REDIS_URL`.
